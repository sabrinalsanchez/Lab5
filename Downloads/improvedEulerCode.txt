% improved euler method 
% solving dy/dx=y 
clear x % Clear values from previous runs
clear y 
h=0.1; % Time step
x(1)=0;
y(1)=1; % initial value
f = @(x,y) y; % here is where you define your rhs 
for i=1:10 % loop to find next x and y 
  x(i+1)=x(i)+h;
  k1=f(x(i),y(i));
  k2=f(x(i)+h,y(i)+h);
  y(i+1)=y(i)+h*(k1+k2)/2; 
end
plot(x,y)
title(['Improved Euler Method using h=', num2str(h),' by JS Bach'])
fprintf('%i\n', y)
